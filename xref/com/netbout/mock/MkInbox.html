27  package com.netbout.mock;
28  
29  import com.jcabi.aspects.Immutable;
30  import com.jcabi.aspects.Loggable;
31  import com.jcabi.jdbc.JdbcSession;
32  import com.jcabi.jdbc.Outcome;
33  import com.jcabi.jdbc.SingleOutcome;
34  import com.netbout.spi.Bout;
35  import com.netbout.spi.Inbox;
36  import com.netbout.spi.Pageable;
37  import java.io.IOException;
38  import java.sql.ResultSet;
39  import java.sql.SQLException;
40  import java.sql.Statement;
41  import java.util.Collection;
42  import java.util.LinkedList;
43  import lombok.EqualsAndHashCode;
44  import lombok.ToString;
45  
46  /**
47   * Cached inbox.
48   *
49   * @author Yegor Bugayenko (yegor@tpc2.com)
50   * @version $Id$
51   * @since 2.4
52   */
53  @Immutable
54  @ToString
55  @Loggable(Loggable.DEBUG)
56  @EqualsAndHashCode(of = { "sql", "self" })
57  final class MkInbox implements Inbox {
58  
59      /**
60       * SQL data source provider.
61       */
62      private final transient Sql sql;
63  
64      /**
65       * Self alias.
66       */
67      private final transient String self;
68  
69      /**
70       * Public ctor.
71       * @param src Source
72       * @param name Alias
73       */
74      MkInbox(final Sql src, final String name) {
75          this.sql = src;
76          this.self = name;
77      }
78  
79      @Override
80      public long start() throws IOException {
81          try {
82              return new JdbcSession(this.sql.source())
83                  .sql("INSERT INTO bout (title) VALUES (?)")
84                  .set("untitled")
85                  .insert(new SingleOutcome<Long>(Long.class));
86          } catch (final SQLException ex) {
87              throw new IOException(ex);
88          }
89      }
90  
91      @Override
92      public long unread() {
93          return 0L;
94      }
95  
96      @Override
97      public Bout bout(final long number) throws Inbox.BoutNotFoundException {
98          final boolean exists;
99          try {
100             exists = new JdbcSession(this.sql.source())
101                 .sql("SELECT number FROM bout WHERE number = ?")
102                 .set(number)
103                 .select(Outcome.NOT_EMPTY);
104         } catch (final SQLException ex) {
105             throw new IllegalStateException(ex);
106         }
107         if (!exists) {
108             throw new Inbox.BoutNotFoundException(number);
109         }
110         return new MkBout(this.sql, number, this.self);
111     }
112 
113     @Override
114     public Pageable<Bout> jump(final long number) {
115         throw new UnsupportedOperationException("#jump()");
116     }
117 
118     @Override
119     public Iterable<Bout> iterate() throws IOException {
120         try {
121             return new JdbcSession(this.sql.source())
122                 // @checkstyle LineLength (1 line)
123                 .sql("SELECT b.number FROM bout b JOIN friend f ON b.number = f.bout WHERE f.alias = ?")
124                 .set(this.self)
125                 .select(
126                     new Outcome<Iterable<Bout>>() {
127                         @Override
128                         public Iterable<Bout> handle(final ResultSet rset,
129                             final Statement stmt) throws SQLException {
130                             final Collection<Bout> list =
131                                 new LinkedList<Bout>();
132                             while (rset.next()) {
133                                 list.add(
134                                     new MkBout(
135                                         MkInbox.this.sql,
136                                         rset.getLong(1),
137                                         MkInbox.this.self
138                                     )
139                                 );
140                             }
141                             return list;
142                         }
143                     }
144                 );
145         } catch (final SQLException ex) {
146             throw new IOException(ex);
147         }
148     }
149 }
