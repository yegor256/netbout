27  package com.netbout.mock;
28  
29  import com.jcabi.aspects.Immutable;
30  import com.jcabi.aspects.Loggable;
31  import com.jcabi.jdbc.JdbcSession;
32  import java.io.File;
33  import java.io.IOException;
34  import java.sql.SQLException;
35  import javax.sql.DataSource;
36  import lombok.EqualsAndHashCode;
37  import lombok.ToString;
38  import org.h2.jdbcx.JdbcDataSource;
39  
40  /**
41   * Mock base.
42   *
43   * @author Yegor Bugayenko (yegor@tpc2.com)
44   * @version $Id$
45   * @since 2.0
46   */
47  @Immutable
48  @ToString
49  @Loggable(Loggable.DEBUG)
50  @EqualsAndHashCode(of = "file")
51  final class H2Sql implements Sql {
52  
53      /**
54       * File with H2 database.
55       */
56      private final transient String file;
57  
58      /**
59       * Public ctor.
60       * @throws IOException If fails
61       */
62      H2Sql() throws IOException {
63          final File tmp = File.createTempFile("netbout-", ".h2");
64          tmp.deleteOnExit();
65          this.file = tmp.getAbsolutePath();
66          final String[] stmts = {
67              // @checkstyle LineLength (5 lines)
68              "CREATE TABLE alias (name VARCHAR, urn VARCHAR, photo VARCHAR)",
69              "CREATE TABLE bout (number BIGINT AUTO_INCREMENT, title VARCHAR, date TIMESTAMP DEFAULT CURRENT_TIMESTAMP)",
70              "CREATE TABLE message (number BIGINT AUTO_INCREMENT, bout BIGINT, text VARCHAR, author VARCHAR, date TIMESTAMP DEFAULT CURRENT_TIMESTAMP)",
71              "CREATE TABLE attachment (name VARCHAR, bout BIGINT, data VARCHAR, author VARCHAR)",
72              "CREATE TABLE friend (alias VARCHAR, bout BIGINT)",
73          };
74          final JdbcSession session = new JdbcSession(this.source());
75          for (final String stmt : stmts) {
76              try {
77                  session.sql(stmt).execute();
78              } catch (final SQLException ex) {
79                  throw new IOException(ex);
80              }
81          }
82      }
83  
84      @Override
85      public DataSource source() {
86          final JdbcDataSource src = new JdbcDataSource();
87          src.setURL(String.format("jdbc:h2:%s", this.file));
88          src.setUser("");
89          src.setPassword("");
90          return src;
91      }
92  }
