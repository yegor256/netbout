27  package com.netbout.client;
28  
29  import com.jcabi.aspects.Immutable;
30  import com.jcabi.aspects.Loggable;
31  import com.jcabi.http.Request;
32  import com.jcabi.http.response.RestResponse;
33  import com.jcabi.http.response.XmlResponse;
34  import com.netbout.spi.Attachments;
35  import com.netbout.spi.Bout;
36  import com.netbout.spi.Friends;
37  import com.netbout.spi.Messages;
38  import java.io.IOException;
39  import java.net.HttpURLConnection;
40  import java.text.ParseException;
41  import java.util.Date;
42  import lombok.EqualsAndHashCode;
43  import org.apache.commons.lang3.time.DateFormatUtils;
44  
45  /**
46   * REST bout.
47   *
48   * @author Yegor Bugayenko (yegor@tpc2.com)
49   * @version $Id$
50   * @since 2.0
51   */
52  @Immutable
53  @Loggable(Loggable.DEBUG)
54  @EqualsAndHashCode(of = { "num", "request" })
55  final class RtBout implements Bout {
56  
57      /**
58       * Its number.
59       */
60      private final transient long num;
61  
62      /**
63       * Request to use.
64       */
65      private final transient Request request;
66  
67      /**
68       * Public ctor.
69       * @param number The number
70       * @param req Request to use
71       */
72      RtBout(final long number, final Request req) {
73          this.num = number;
74          this.request = req;
75      }
76  
77      @Override
78      public String toString() {
79          return Long.toString(this.num);
80      }
81  
82      @Override
83      public long number() {
84          return this.num;
85      }
86  
87      @Override
88      public Date date() throws IOException {
89          try {
90              return DateFormatUtils.ISO_DATETIME_FORMAT.parse(
91                  this.request.fetch()
92                      .as(RestResponse.class)
93                      .assertStatus(HttpURLConnection.HTTP_OK)
94                      .as(XmlResponse.class)
95                      .xml()
96                      .xpath("/page/bout/date/text()")
97                      .get(0)
98              );
99          } catch (final ParseException ex) {
100             throw new IllegalStateException(ex);
101         }
102     }
103 
104     @Override
105     public Date updated() {
106         throw new UnsupportedOperationException("#updated()");
107     }
108 
109     @Override
110     public String title() throws IOException {
111         return this.request.fetch()
112             .as(RestResponse.class)
113             .assertStatus(HttpURLConnection.HTTP_OK)
114             .as(XmlResponse.class)
115             .xml()
116             .xpath("/page/bout/title/text()")
117             .get(0);
118     }
119 
120     @Override
121     public void rename(final String text) throws IOException {
122         this.request.fetch()
123             .as(RestResponse.class)
124             .assertStatus(HttpURLConnection.HTTP_OK)
125             .as(XmlResponse.class)
126             .rel("/page/links/link[@rel='rename']/@href")
127             .method(Request.POST)
128             .body().formParam("title", text).back()
129             .fetch()
130             .as(RestResponse.class)
131             .assertStatus(HttpURLConnection.HTTP_SEE_OTHER);
132     }
133 
134     @Override
135     public Messages messages() {
136         return new RtMessages(this.request);
137     }
138 
139     @Override
140     public Friends friends() {
141         return new RtFriends(this.request);
142     }
143 
144     @Override
145     public Attachments attachments() {
146         return new RtAttachments(this.request);
147     }
148 }
