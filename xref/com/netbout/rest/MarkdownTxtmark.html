27  package com.netbout.rest;
28  
29  import com.github.rjeschke.txtmark.Configuration;
30  import com.github.rjeschke.txtmark.Processor;
31  import java.util.regex.Matcher;
32  import java.util.regex.Pattern;
33  import javax.validation.constraints.NotNull;
34  
35  /**
36   * Text with markdown formatting.
37   * Using TxtMark markdown processor.
38   *
39   * @author Dmitry Zaytsev (dmitry.zaytsev@gmail.com)
40   * @version $Id: a97263f7f82529596ebe13e167e0e50f891be152 $
41   * @since 2.23
42   */
43  public final class MarkdownTxtmark implements Markdown {
44      /**
45       * Plain link detection pattern.
46       */
47      private static final Pattern LINK = Pattern.compile(
48          // @checkstyle LineLengthCheck (1 line)
49          "(?<!\\]\\s{0,256}\\()(?<!\\]:\\s{0,256})(?<!=\")(https?:\\/\\/[a-zA-Z0-9-._~:\\?#@!$&'*+,;=%\\/]+[a-zA-Z0-9-_~#@$&'*+=%\\/])(?![\\w.]*\\]\\()"
50      );
51      /**
52       * Pattern to detect lines which should have a line break at the end.
53       * We look for lines which have less than two spaces on their end because
54       * TxtMark automatically puts {@code <br/>} if the line has two or more
55       * spaces on its end and we should skip a line that starts with four
56       * or more spaces because it's a code block in markdown.
57       */
58      private static final Pattern NEW_LINE = Pattern.compile(
59          "^ {0,3}(\\S|(\\S.*\\S)) ?$", Pattern.MULTILINE
60      );
61  
62      @Override
63      public String html(@NotNull final String txt) {
64          final Configuration conf = Configuration.builder()
65              .enableSafeMode().build();
66          return Processor.process(
67              MarkdownTxtmark.formatLinks(MarkdownTxtmark.makeLineBreak(txt)),
68              conf
69          );
70      }
71  
72      /**
73       * Replace plain links with Markdown syntax. To be convinced it doesn't
74       * replace links inside markdown syntax, it ensures that characters
75       * before and after link do not match to Markdown link syntax.
76       * @param txt Text to find links in
77       * @return Text with Markdown-formatted links
78       */
79      private static String formatLinks(final String txt) {
80          final StringBuffer result = new StringBuffer();
81          final Matcher matcher = MarkdownTxtmark.LINK.matcher(txt);
82          while (matcher.find()) {
83              matcher.appendReplacement(
84                  result,
85                  String.format("[%1$s](%1$s)", matcher.group())
86              );
87          }
88          matcher.appendTail(result);
89          return result.toString();
90      }
91      /**
92       * Insert two spaces at the end of string that needs line break to
93       * force creation of HTML line break.
94       * @param txt Text to replace
95       * @return Text with Markdown-formatted links
96       */
97      private static String makeLineBreak(final String txt) {
98          final StringBuffer result = new StringBuffer();
99          final Matcher matcher = MarkdownTxtmark.NEW_LINE.matcher(txt);
100         while (matcher.find()) {
101             if (!matcher.hitEnd()) {
102                 matcher.appendReplacement(
103                     result,
104                     String.format(
105                         "%s  ", Matcher.quoteReplacement(matcher.group())
106                     )
107                 );
108             }
109         }
110         matcher.appendTail(result);
111         return result.toString();
112     }
113 }
