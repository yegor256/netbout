56  /**
57   * Authenticated app.
58   *
59   * @author Yegor Bugayenko (yegor@teamed.io)
60   * @version $Id$
61   * @since 2.14
62   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
63   */
64  final class TkAppAuth extends TkWrap {
65  
66      /**
67       * Testing mode is ON?
68       */
69      private static final boolean TESTING =
70          Manifests.read("Netbout-DynamoKey").startsWith("AAAA");
71  
72      /**
73       * Ctor.
74       * @param take Take
75       */
76      TkAppAuth(final Take take) {
77          super(TkAppAuth.make(take));
78      }
79  
80      /**
81       * Authenticated.
82       * @param take Take
83       * @return Authenticated take
84       */
85      private static Take make(final Take take) {
86          return new TkAuth(
87              take,
88              new PsChain(
89                  new PsByFlag(
90                      new PsByFlag.Pair(
91                          PsGithub.class.getSimpleName(),
92                          new PsGithub(
93                              Manifests.read("Netbout-GithubId"),
94                              Manifests.read("Netbout-GithubSecret")
95                          )
96                      ),
97                      new PsByFlag.Pair(
98                          PsFacebook.class.getSimpleName(),
99                          new PsFacebook(
100                             Manifests.read("Netbout-FbId"),
101                             Manifests.read("Netbout-FbSecret")
102                         )
103                     ),
104                     new PsByFlag.Pair(
105                         PsGoogle.class.getSimpleName(),
106                         new PsGoogle(
107                             Manifests.read("Netbout-GoogleId"),
108                             Manifests.read("Netbout-GoogleSecret"),
109                             "http://www.netbout.com/?PsByFlag=PsGoogle"
110                         )
111                     ),
112                     new PsByFlag.Pair(
113                         "fake-user",
114                         new TkAppAuth.FakePass()
115                     ),
116                     new PsByFlag.Pair(
117                         PsLogout.class.getSimpleName(),
118                         new PsLogout()
119                     )
120                 ),
121                 new PsCookie(
122                     new CcSafe(
123                         new CcHex(
124                             new CcXOR(
125                                 new CcSalted(new CcCompact()),
126                                 Manifests.read("Netbout-SecurityKey")
127                             )
128                         )
129                     )
130                 ),
131                 new PsFake(TkAppAuth.TESTING)
132             )
133         );
134     }
135 
136     /**
137      * Fake pass.
138      */
139     private static final class FakePass implements Pass {
140         @Override
141         public Iterator<Identity> enter(final Request req) throws IOException {
142             final Collection<Identity> user = new ArrayList<>(1);
143             if (TkAppAuth.TESTING) {
144                 user.add(
145                     new Identity.Simple(
146                         new RqHref.Smart(new RqHref.Base(req)).single("urn")
147                     )
148                 );
149             }
150             return user.iterator();
151         }
152         @Override
153         public Response exit(final Response response, final Identity identity) {
154             return response;
155         }
156     }
157 }
