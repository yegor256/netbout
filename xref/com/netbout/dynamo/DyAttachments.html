27  package com.netbout.dynamo;
28  
29  import com.google.common.base.Function;
30  import com.google.common.base.Predicates;
31  import com.google.common.collect.Iterables;
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.aspects.Tv;
35  import com.jcabi.dynamo.Attributes;
36  import com.jcabi.dynamo.Conditions;
37  import com.jcabi.dynamo.Item;
38  import com.jcabi.dynamo.QueryValve;
39  import com.jcabi.dynamo.Region;
40  import com.netbout.spi.Attachment;
41  import com.netbout.spi.Attachments;
42  import java.io.IOException;
43  import java.util.Iterator;
44  import lombok.EqualsAndHashCode;
45  import lombok.ToString;

47  /**
48   * Dynamo attachments.
49   *
50   * @author Yegor Bugayenko (yegor@teamed.io)
51   * @version $Id$
52   * @since 2.0
53   */
54  @Immutable
55  @Loggable(Loggable.DEBUG)
56  @ToString(of = "bout")
57  @EqualsAndHashCode(of = { "region", "bout", "self" })
58  final class DyAttachments implements Attachments {
59  
60      /**
61       * Table name.
62       */
63      public static final String TBL = "attachments";
64  
65      /**
66       * Bout attribute.
67       */
68      public static final String HASH = "bout";
69  
70      /**
71       * Attachment name attribute.
72       */
73      public static final String RANGE = "name";
74  
75      /**
76       * Owner of the attachment.
77       */
78      public static final String ATTR_ALIAS = "alias";
79  
80      /**
81       * Ctype of the attachment.
82       */
83      public static final String ATTR_CTYPE = "ctype";
84  
85      /**
86       * Etag of the attachment.
87       */
88      public static final String ATTR_ETAG = "etag";
89  
90      /**
91       * Data.
92       */
93      public static final String ATTR_DATA = "data";
94  
95      /**
96       * S3 key.
97       */
98      public static final String ATTR_KEY = "s3";
99  
100     /**
101      * Region to work with.
102      */
103     private final transient Region region;
104 
105     /**
106      * Bout number.
107      */
108     private final transient long bout;
109 
110     /**
111      * Self alias.
112      */
113     private final transient String self;
114 
115     /**
116      * Ctor.
117      * @param reg Region
118      * @param num Bout number
119      * @param slf Self
120      */
121     DyAttachments(final Region reg, final long num, final String slf) {
122         this.region = reg;
123         this.bout = num;
124         this.self = slf;
125     }
126 
127     @Override
128     public Attachment get(final String name)
129         throws Attachments.NotFoundException {
130         final Iterator<Item> items = this.region.table(DyAttachments.TBL)
131             .frame()
132             .through(
133                 new QueryValve()
134                     .withLimit(1)
135                     .withAttributesToGet(
136                         DyAttachments.ATTR_CTYPE,
137                         DyAttachments.ATTR_ETAG,
138                         DyAttachments.ATTR_KEY
139                     )
140             )
141             .where(DyAttachments.HASH, Conditions.equalTo(this.bout))
142             .where(DyAttachments.RANGE, name)
143             .iterator();
144         if (!items.hasNext()) {
145             throw new Attachments.NotFoundException(
146                 String.format("attachment '%s' not found", name)
147             );
148         }
149         return new DyAttachment(this.region, items.next(), this.self);
150     }
151 
152     @Override
153     public Iterable<Attachment> iterate() {
154         return Iterables.transform(
155             this.region.table(DyAttachments.TBL)
156                 .frame()
157                 .through(
158                     new QueryValve().withAttributesToGet(
159                         DyAttachments.ATTR_ALIAS,
160                         DyAttachments.ATTR_CTYPE,
161                         DyAttachments.ATTR_ETAG
162                     )
163                 )
164                 .where(DyAttachments.HASH, Conditions.equalTo(this.bout)),
165             new Function<Item, Attachment>() {
166                 @Override
167                 public Attachment apply(final Item item) {
168                     return new DyAttachment(
169                         DyAttachments.this.region,
170                         item, DyAttachments.this.self
171                     );
172                 }
173             }
174         );
175     }
176 
177     @Override
178     public int unseen() throws IOException {
179         final Item itm = this.region.table(DyFriends.TBL)
180             .frame()
181             .through(
182                 new QueryValve()
183                     .withLimit(1)
184                     .withAttributesToGet(DyFriends.ATTR_UNSEEN)
185             )
186             .where(DyFriends.HASH, Conditions.equalTo(this.bout))
187             .where(DyFriends.RANGE, Conditions.equalTo(this.self))
188             .iterator().next();
189         final int unseen;
190         if (itm.has(DyFriends.ATTR_UNSEEN)) {
191             unseen = itm.get(DyFriends.ATTR_UNSEEN).getSS().size();
192         } else {
193             unseen = 0;
194         }
195         return unseen;
196     }
197 
198     @Override
199     public void create(final String name) throws IOException {
200         if (!name.matches("[a-zA-Z\\.\\-0-9]{3,100}")) {
201             throw new Attachments.InvalidNameException(
202                 String.format("invalid attachment name '%s'", name)
203             );
204         }
205         this.region.table(DyAttachments.TBL).put(
206             new Attributes()
207                 .with(DyAttachments.HASH, this.bout)
208                 .with(DyAttachments.RANGE, name)
209                 .with(DyAttachments.ATTR_ALIAS, this.self)
210                 .with(DyAttachments.ATTR_CTYPE, "text/plain")
211                 .with(DyAttachments.ATTR_DATA, " ")
212                 .with(DyAttachments.ATTR_ETAG, "empty")
213         );
214     }
215 
216     @Override
217     public void delete(final String name) {
218         if (name.isEmpty()) {
219             throw new IllegalArgumentException("name can't be empty");
220         }
221         if (name.length() > Tv.HUNDRED) {
222             throw new IllegalArgumentException("name is too long");
223         }
224         Iterables.removeIf(
225             this.region.table(DyAttachments.TBL)
226                 .frame()
227                 .where(DyAttachments.HASH, Conditions.equalTo(this.bout))
228                 .where(DyAttachments.RANGE, name),
229             Predicates.alwaysTrue()
230         );
231     }
232 
233 }
