61  /**
62   * Dynamo attachment.
63   *
64   * @author Yegor Bugayenko (yegor@teamed.io)
65   * @version $Id: 7e53027b09017ff69e3968f30f0157fe62492abb $
66   * @since 2.0
67   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
68   */
69  @Immutable
70  @Loggable(Loggable.DEBUG)
71  @ToString(of = "item")
72  @EqualsAndHashCode(of = { "region", "item", "self" })
73  @SuppressWarnings({ "PMD.TooManyMethods", "PMD.ExcessiveImports" })
74  final class DyAttachment implements Attachment {
75  
76      /**
77       * Maximum size to store in DynamoDB.
78       */
79      private static final int MAX_SIZE = 50000;
80  
81      /**
82       * Bucket with attachments.
83       */
84      private final transient Bucket bucket;
85  
86      /**
87       * Region we're in.
88       */
89      private final transient Region region;
90  
91      /**
92       * Item with data.
93       */
94      private final transient Item item;
95  
96      /**
97       * Self alias.
98       */
99      private final transient String self;

101     /**
102      * Ctor.
103      * @param reg Region
104      * @param itm Item
105      * @param slf Self alias
106      */
107     DyAttachment(final Region reg, final Item itm, final String slf) {
108         this(reg, DyAttachment.storage(), itm, slf);
109     }
110 
111     /**
112      * Ctor.
113      * @param reg Region
114      * @param bkt Bucket
115      * @param itm Item
116      * @param slf Self alias
117      * @since 2.8
118      * @checkstyle ParameterNumberCheck (5 lines)
119      */
120     DyAttachment(final Region reg, final Bucket bkt,
121         final Item itm, final String slf) {
122         this.region = reg;
123         this.bucket = bkt;
124         this.item = itm;
125         this.self = slf;
126     }
127 
128     @Override
129     public String name() throws IOException {
130         return this.item.get(DyAttachments.RANGE).getS();
131     }
132 
133     @Override
134     public String ctype() throws IOException {
135         return this.item.get(DyAttachments.ATTR_CTYPE).getS();
136     }
137 
138     @Override
139     public String etag() throws IOException {
140         final String etag;
141         if (this.item.has(DyAttachments.ATTR_ETAG)) {
142             etag = this.item.get(DyAttachments.ATTR_ETAG).getS();
143         } else {
144             etag = "";
145         }
146         return etag;
147     }
148 
149     @Override
150     public boolean unseen() throws IOException {
151         final Item itm = this.region.table(DyFriends.TBL)
152             .frame()
153             .through(
154                 new QueryValve()
155                     .withLimit(1)
156                     .withAttributesToGet(DyFriends.ATTR_UNSEEN)
157             )
158             .where(DyFriends.HASH, Conditions.equalTo(this.bout()))
159             .where(DyFriends.RANGE, Conditions.equalTo(this.self))
160             .iterator().next();
161         return itm.has(DyFriends.ATTR_UNSEEN)
162             && itm.get(DyFriends.ATTR_UNSEEN).getSS().contains(this.name());
163     }
164 
165     @Override
166     public InputStream read() throws IOException {
167         this.seen();
168         final byte[] bytes;
169         if (this.item.has(DyAttachments.ATTR_KEY)) {
170             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
171             this.bucket.ocket(
172                 this.item.get(DyAttachments.ATTR_KEY).getS()
173             ).read(baos);
174             bytes = baos.toByteArray();
175         } else {
176             final ByteBuffer buf =
177                 this.item.get(DyAttachments.ATTR_DATA).getB();
178             if (buf == null) {
179                 bytes = "temporary unavailable".getBytes(
180                     StandardCharsets.UTF_8
181                 );
182             } else {
183                 bytes = buf.array();
184             }
185         }
186         return new ByteArrayInputStream(bytes);
187     }
188 
189     @Override
190     public void write(final InputStream stream, final String ctype,
191         final String etag) throws IOException {
192         final byte[] data = IOUtils.toByteArray(stream);
193         if (!etag.equals(this.etag())) {
194             final AttributeUpdates updates = this.save(data, ctype)
195                 .with(DyAttachments.ATTR_ETAG, etag);
196             this.item.put(updates);
197             this.updated();
198             Logger.info(
199                 this, "attachment %s (%d bytes) uploaded into #%d by @%s",
200                 this.name(), data.length, this.bout(), this.self
201             );
202         }
203     }
204 
205     /**
206      * Save content.
207      * @param data Data to save
208      * @param ctype CType
209      * @return Updates to apply
210      * @throws IOException If fails
211      */
212     private AttributeUpdates save(final byte[] data, final String ctype)
213         throws IOException {
214         AttributeUpdates updates = new AttributeUpdates()
215             .with(DyAttachments.ATTR_CTYPE, ctype);
216         if (data.length < DyAttachment.MAX_SIZE) {
217             if (this.item.has(DyAttachments.ATTR_KEY)) {
218                 this.bucket.remove(
219                     this.item.get(DyAttachments.ATTR_KEY).getS()
220                 );
221             }
222             updates = updates.with(
223                 DyAttachments.ATTR_DATA,
224                 new AttributeValue().withB(ByteBuffer.wrap(data))
225             );
226             updates = updates.with(
227                 DyAttachments.ATTR_KEY,
228                 new AttributeValueUpdate().withAction(
229                     AttributeAction.DELETE
230                 )
231             );
232         } else {
233             final String key;
234             if (this.item.has(DyAttachments.ATTR_KEY)) {
235                 key = this.item.get(DyAttachments.ATTR_KEY).getS();
236             } else {
237                 key = String.format(
238                     "%d/%s",
239                     Long.parseLong(
240                         this.item.get(DyAttachments.HASH).getN()
241                     ),
242                     this.item.get(DyAttachments.RANGE).getS()
243                 );
244             }
245             final ObjectMetadata meta = new ObjectMetadata();
246             meta.setContentType(ctype);
247             meta.setContentLength((long) data.length);
248             this.bucket.ocket(key).write(
249                 new ByteArrayInputStream(data),
250                 meta
251             );
252             updates = updates
253                 .with(DyAttachments.ATTR_DATA, key)
254                 .with(DyAttachments.ATTR_KEY, key);
255         }
256         return updates;
257     }
258 
259     /**
260      * Bout we're in.
261      * @return Bout number
262      * @throws IOException If fails
263      */
264     private long bout() throws IOException {
265         return Long.parseLong(this.item.get(DyAttachments.HASH).getN());
266     }

268     /**
269      * It was updated just now.
270      * @throws IOException If fails
271      */
272     @Async
273     private void updated() throws IOException {
274         Iterables.all(
275             this.region.table(DyFriends.TBL).frame()
276                 .through(new QueryValve())
277                 .where(DyFriends.HASH, Conditions.equalTo(this.bout())),
278             // @checkstyle AnonInnerLengthCheck (50 lines)
279             new Predicate<Item>() {
280                 @Override
281                 public boolean apply(final Item input) {
282                     AttributeUpdates updates = new AttributeUpdates().with(
283                         DyFriends.ATTR_UPDATED,
284                         System.currentTimeMillis()
285                     );
286                     try {
287                         final String alias = input.get(DyFriends.RANGE).getS();
288                         if (!alias.equals(DyAttachment.this.self)) {
289                             final Set<String> list =
290                                 DyAttachment.this.list(alias);
291                             list.add(DyAttachment.this.name());
292                             updates = updates.with(
293                                 DyFriends.ATTR_UNSEEN,
294                                 new AttributeValueUpdate()
295                                     .withAction(AttributeAction.PUT)
296                                     .withValue(
297                                         new AttributeValue().withSS(list)
298                                     )
299                             );
300                         }
301                         input.put(updates);
302                     } catch (final IOException ex) {
303                         throw new IllegalStateException(ex);
304                     }
305                     return true;
306                 }
307             }
308         );
309     }

311     /**
312      * It was seen just now.
313      * @throws IOException If fails
314      */
315     @Async
316     private void seen() throws IOException {
317         Iterables.all(
318             this.region.table(DyFriends.TBL).frame()
319                 .through(new QueryValve())
320                 .where(DyFriends.HASH, Conditions.equalTo(this.bout()))
321                 .where(DyFriends.RANGE, this.self),
322             new Predicate<Item>() {
323                 @Override
324                 public boolean apply(final Item input) {
325                     try {
326                         final Set<String> list =
327                             DyAttachment.this.list(DyAttachment.this.self);
328                         list.remove(DyAttachment.this.name());
329                         if (list.isEmpty()) {
330                             input.put(
331                                 new AttributeUpdates().with(
332                                     DyFriends.ATTR_UNSEEN,
333                                     new AttributeValueUpdate().withAction(
334                                         AttributeAction.DELETE
335                                     )
336                                 )
337                             );
338                         } else {
339                             input.put(
340                                 new AttributeUpdates().with(
341                                     DyFriends.ATTR_UNSEEN,
342                                     new AttributeValueUpdate()
343                                         .withAction(AttributeAction.PUT)
344                                         .withValue(
345                                             new AttributeValue().withSS(list)
346                                         )
347                                 )
348                             );
349                         }
350                     } catch (final IOException ex) {
351                         throw new IllegalStateException(ex);
352                     }
353                     return true;
354                 }
355             }
356         );
357     }

359     /**
360      * Existing list of unseen attachments.
361      * @param alias Alias
362      * @return List of attachments that are unseen
363      * @throws IOException If fails
364      */
365     private Set<String> list(final String alias) throws IOException {
366         final Item itm = this.region.table(DyFriends.TBL)
367             .frame().through(new QueryValve())
368             .where(DyFriends.HASH, Conditions.equalTo(this.bout()))
369             .where(DyFriends.RANGE, Conditions.equalTo(alias))
370             .iterator().next();
371         final Set<String> list = new HashSet<String>(0);
372         if (itm.has(DyFriends.ATTR_UNSEEN)) {
373             list.addAll(itm.get(DyFriends.ATTR_UNSEEN).getSS());
374         }
375         return list;
376     }

378     /**
379      * S3 Bucket storage.
380      * @return Bucket
381      */
382     @Cacheable(forever = true)
383     private static Bucket storage() {
384         final String key = Manifests.read("Netbout-S3Key");
385         final com.jcabi.s3.Region region;
386         if (key.matches("[0-9A-Z]{20}")) {
387             region = new com.jcabi.s3.Region.Simple(
388                 key,
389                 Manifests.read("Netbout-S3Secret")
390             );
391         } else {
392             region = new MkRegion();
393         }
394         return region.bucket(Manifests.read("Netbout-S3Bucket"));
395     }
