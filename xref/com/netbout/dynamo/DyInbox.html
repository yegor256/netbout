58  /**
59   * Dynamo inbox.
60   *
61   * @author Yegor Bugayenko (yegor@teamed.io)
62   * @version $Id$
63   * @since 2.0
64   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
65   */
66  @Immutable
67  @Loggable(Loggable.DEBUG)
68  @ToString(of = "self")
69  @EqualsAndHashCode(of = { "counter", "region", "self", "since" })
70  final class DyInbox implements Inbox {
71  
72      /**
73       * Counter with bout number.
74       */
75      private final transient Counter counter;
76  
77      /**
78       * Region we're in.
79       */
80      private final transient Region region;
81  
82      /**
83       * Alias of myself.
84       */
85      private final transient String self;
86  
87      /**
88       * Start moment.
89       */
90      private final transient long since;
91  
92      /**
93       * Ctor.
94       * @param reg Region we're in
95       * @param slf My alias
96       */
97      DyInbox(final Region reg, final String slf) {
98          this(reg, slf, DyInbox.sttc(), Inbox.NEVER);
99      }
100 
101     /**
102      * Ctor.
103      * @param reg Region we're in
104      * @param slf My alias
105      * @param ctr Counter
106      * @param strt Start
107      * @since 2.7.1
108      * @checkstyle ParameterNumberCheck (5 lines)
109      */
110     DyInbox(final Region reg, final String slf, final Counter ctr,
111         final long strt) {
112         this.region = reg;
113         this.self = slf;
114         this.counter = ctr;
115         this.since = strt;
116     }
117 
118     @Override
119     public long start() throws IOException {
120         final long number = this.counter.incrementAndGet(1L);
121         this.region.table(DyFriends.TBL).put(
122             new Attributes()
123                 .with(DyFriends.RANGE, this.self)
124                 .with(DyFriends.HASH, number)
125                 .with(DyFriends.ATTR_UPDATED, System.currentTimeMillis())
126                 .with(DyFriends.ATTR_TITLE, "untitled")
127         );
128         Logger.info(this, "bout #%d started by @%s", number, this.self);
129         return number;
130     }
131 
132     /**
133      * {@inheritDoc}
134      * @todo #1 DynamoDBLocal doesn't work with all attributes
135      *  in global index. If we remove this check for a local version,
136      *  most tests fail. I'm not sure how to fix that. I suspect, it's
137      *  a bug in DynamoDBLocal, but I don't even know how to report
138      *  it to them :( Anyway, let's try to investigate and either fix
139      *  property or introduce a better workaround. Pay attention that this
140      *  code works correctly in production.
141      */
142     @Override
143     public long unread() throws IOException {
144         long unread = 0L;
145         if (!"1.0-LOCAL".equals(Manifests.read("Netbout-Version"))) {
146             final Iterable<Item> items = this.region.table(DyFriends.TBL)
147                 .frame()
148                 .where(DyFriends.RANGE, this.self)
149                 .through(
150                     new QueryValve()
151                         .withIndexName(DyFriends.INDEX)
152                         .withConsistentRead(false)
153                         .withSelect(Select.SPECIFIC_ATTRIBUTES)
154                         .withAttributesToGet(DyFriends.ATTR_UNREAD)
155                         .withScanIndexForward(false)
156                 );
157             for (final Item item : items) {
158                 if (item.has(DyFriends.ATTR_UNREAD)) {
159                     unread += Long.parseLong(
160                         item.get(DyFriends.ATTR_UNREAD).getN()
161                     );
162                 }
163             }
164         }
165         return unread;
166     }
167 
168     @Override
169     @Loggable(
170         value = Loggable.DEBUG,
171         ignore = Inbox.BoutNotFoundException.class
172     )
173     public Bout bout(final long number) throws Inbox.BoutNotFoundException {
174         final Iterator<Item> items = this.region.table(DyFriends.TBL)
175             .frame()
176             .through(
177                 new QueryValve().withLimit(1)
178                     .withSelect(Select.SPECIFIC_ATTRIBUTES)
179                     .withAttributesToGet(
180                         DyFriends.HASH, DyFriends.RANGE
181                     )
182             )
183             .where(DyFriends.HASH, Conditions.equalTo(number))
184             .where(DyFriends.RANGE, this.self)
185             .iterator();
186         if (!items.hasNext()) {
187             throw new Inbox.BoutNotFoundException(number);
188         }
189         return new DyBout(this.region, items.next(), this.self);
190     }
191 
192     @Override
193     public Pageable<Bout> jump(final long number) {
194         return new DyInbox(this.region, this.self, this.counter, number);
195     }
196 
197     @Override
198     public Iterable<Bout> iterate() {
199         return Iterables.transform(
200             this.region.table(DyFriends.TBL)
201                 .frame()
202                 .where(DyFriends.RANGE, this.self)
203                 .where(
204                     DyFriends.ATTR_UPDATED,
205                     new Condition()
206                         .withComparisonOperator(ComparisonOperator.LT)
207                         .withAttributeValueList(
208                             new AttributeValue().withN(
209                                 Long.toString(this.since)
210                             )
211                         )
212                 )
213                 .through(
214                     new QueryValve()
215                         .withIndexName(DyFriends.INDEX)
216                         .withConsistentRead(false)
217                         .withLimit(Inbox.PAGE)
218                         .withSelect(Select.ALL_PROJECTED_ATTRIBUTES)
219                         .withScanIndexForward(false)
220                 ),
221             new Function<Item, Bout>() {
222                 @Override
223                 public Bout apply(final Item item) {
224                     return new Bout.ReadOnly(
225                         new DyBout(
226                             DyInbox.this.region,
227                             item, DyInbox.this.self
228                         )
229                     );
230                 }
231             }
232         );
233     }
234 
235     /**
236      * Sttc counter.
237      * @return Counter
238      */
239     @Cacheable(forever = true)
240     private static Counter sttc() {
241         try {
242             return new CdSttc(
243                 new ReSttc(
244                     RtSttc.make(
245                         URN.create(Manifests.read("Netbout-SttcUrn")),
246                         Manifests.read("Netbout-SttcToken")
247                     )
248                 )
249             ).counters().get("nb-bout");
250         } catch (final IOException ex) {
251             throw new IllegalStateException(ex);
252         }
253     }
